  <div class="container">
    <h1>E-Consultation Feedback</h1>
    
    <form id="feedbackForm">
      <label for="name">Name</label>
      <input type="text" id="name" placeholder="Enter your name" required>
      
      <label for="stakeholder">Stakeholder Type</label>
      <select id="stakeholder" required>
        <option value="">Select type</option>
        <option value="Citizen">Citizen</option>
        <option value="NGO">NGO</option>
        <option value="Industry">Industry</option>
        <option value="Academia">Academia</option>
      </select>
      
      <label for="comment">Your Comment</label>
      <textarea id="comment" rows="5" placeholder="Share your views on the draft policy..." maxlength="500" required></textarea>
      <div class="counter"><span id="charCount">0</span>/500</div>
      
      <button type="submit">Submit Feedback</button>
    </form>
    
    <div id="output"></div>
  </div>

  <script>
    // JavaScript starts here
    const form = document.getElementById('feedbackForm');
    const comment = document.getElementById('comment');
    const charCount = document.getElementById('charCount');
    const output = document.getElementById('output');

    // Live character counter
    comment.addEventListener('input', () => {
      charCount.textContent = comment.value.length;
      
      if (comment.value.length > 450) {
        charCount.style.color = '#dc2626';
      } else {
        charCount.style.color = '#64748b';
      }
    });

    // Form submission
    form.addEventListener('submit', (e) => {
      e.preventDefault();
      
      const name = document.getElementById('name').value;
      const stakeholder = document.getElementById('stakeholder').value;
      const commentText = comment.value;
      
      // Simple client-side validation
      if (commentText.length < 10) {
        alert('Please enter at least 10 characters for your comment.');
        return;
      }
      
      // Show confirmation - in real app you'd POST to your E-consultation backend
      output.style.display = 'block';
      output.innerHTML = `
        <strong>Thank you, ${name}!</strong><br>
        Your feedback as <em>${stakeholder}</em> has been recorded.<br>
        <br>
        <strong>Comment:</strong> ${commentText}
      `;
      
      // Reset form
      form.reset();
      charCount.textContent = '0';
      
      // Scroll to output
      output.scrollIntoView({ behavior: 'smooth' });
    });
  </script>
